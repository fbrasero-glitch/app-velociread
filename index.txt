      // PDF text extraction using PDF.js
      async function parsePDF(arrayBuffer) {
        if (typeof pdfjsLib === 'undefined') {

        const loadingTask = pdfjsLib.getDocument({ data: arrayBuffer });

          const page = await pdf.getPage(pageNum);
          const textContent = await page.getTextContent();
          
          let lastY = -1;
          let pageText = "";

            lastY = item.transform[5];

        }
        return fullText;

      // EPUB parser using JSZip and DOMParser
      async function parseEPUB(arrayBuffer) {
        if (typeof JSZip === 'undefined') {
          throw new Error("Librería JSZip no disponible.");
        }

        const zip = await JSZip.loadAsync(arrayBuffer);
        
        // 1. Encuentra container.xml
        const containerFile = zip.file("META-INF/container.xml");
        if (!containerFile) throw new Error("No es un EPUB válido (falta container.xml)");
        
        const containerXmlStr = await containerFile.async("text");
        const parser = new DOMParser();
        const containerDoc = parser.parseFromString(containerXmlStr, "text/xml");
        const rootfile = containerDoc.querySelector("rootfile");
        if (!rootfile) throw new Error("No es un EPUB válido (falta rootfile en container.xml)");
        
        const opfPath = rootfile.getAttribute("full-path");
        
        // Carpeta base del archivo OPF para mapear rutas relativas
        let opfDir = "";
        if (opfPath.includes("/")) {
          opfDir = opfPath.substring(0, opfPath.lastIndexOf("/")) + "/";
        }
        
        const resolvePath = (relativeHref) => {
          let decodedHref = decodeURIComponent(relativeHref);
          if (decodedHref.startsWith("/")) return decodedHref.substring(1);
          
          const combined = opfDir + decodedHref;
          const parts = combined.split("/");
          const resolved = [];
          for (let part of parts) {
            if (part === "." || part === "") continue;
            if (part === "..") {
              resolved.pop();
            } else {
              resolved.push(part);
            }
          }
          return resolved.join("/");
        };

        // 2. Lee el manifiesto OPF
        const opfFile = zip.file(opfPath);
        if (!opfFile) throw new Error("No se encuentra el archivo OPF");
        
        const opfXmlStr = await opfFile.async("text");
        const opfDoc = parser.parseFromString(opfXmlStr, "text/xml");
        
        // 3. Mapea items del manifest
        const manifestItems = {};
        opfDoc.querySelectorAll("manifest > item").forEach(item => {
          const id = item.getAttribute("id");
          const href = item.getAttribute("href");
          const mediaType = item.getAttribute("media-type");
          manifestItems[id] = { href, mediaType };
        });
        
        // 4. Lee el spine (orden de lectura)
        const spineIds = Array.from(opfDoc.querySelectorAll("spine > itemref")).map(ref => {
          return ref.getAttribute("idref");
        });
        
        // 5. Extrae textos XHTML
        let combinedText = "";
        for (let idref of spineIds) {
          const item = manifestItems[idref];
          if (item && (item.mediaType.includes("html") || item.mediaType.includes("xml"))) {
            const zipPath = resolvePath(item.href);
            const zipFile = zip.file(zipPath);
            if (zipFile) {
              const htmlStr = await zipFile.async("text");
              const htmlDoc = parser.parseFromString(htmlStr, "text/html");
              
              // Elimina código, estilos y metadata
              htmlDoc.querySelectorAll("script, style, head, svg, nav").forEach(el => el.remove());
              
              const text = htmlDoc.body ? htmlDoc.body.textContent : htmlDoc.documentElement.textContent;
              if (text) {
                combinedText += text.trim() + "\n\n";
              }
            }
          }
        }
        
        if (!combinedText.trim()) {
          throw new Error("No se pudo extraer texto del EPUB.");
        }
        return combinedText;
      }

        let finalDelay = baseDelay;

        // Check clean punctuation mapping
        if (/[.!?]$/.test(word)) {
          finalDelay = baseDelay * 2.2;
        } else if (/[;:,]$/.test(word)) {
          finalDelay = baseDelay * 1.5;
        }

        // Cumulative length constraint (greater than 10 characters)
        if (word.length > 10) {
          finalDelay = finalDelay * 1.3;
        }

        return finalDelay;
      }

      // ORP index calculation
      function getOrpIndexForWord(word) {
        const len = word.length;
        if (len === 1) return 0;
        if (len >= 2 && len <= 5) return 1;
        if (len >= 6 && len <= 9) return 2;
        if (len >= 10 && len <= 13) return 3;
        return 4; // Length > 13
      }

      // Render the current word split by ORP character
      function renderActiveWord() {
        if (state.words.length === 0) return;

        const currentWord = state.words[state.currentIndex];
        const orpIdx = getOrpIndexForWord(currentWord);

        const leftStr = currentWord.substring(0, orpIdx);
        const orpChar = currentWord.charAt(orpIdx) || '';
        const rightStr = currentWord.substring(orpIdx + 1);

        el.wordLeft.textContent = leftStr;
        el.wordOrp.textContent = orpChar;
        el.wordRight.textContent = rightStr;
      }

      // Update progress fill and numerical stats labels
      function updateProgressUI() {
        const total = state.words.length;
        const current = total > 0 ? state.currentIndex + 1 : 0;
        
        const pct = total > 0 ? (current / total) * 100 : 0;
        el.readingProgressFill.style.width = `${pct}%`;

        el.readingStatsTxt.textContent = `${current} / ${total}`;
      }

      // Loop manager: setTimeout implementation
      function playRSVPEngine() {
        if (state.currentIndex >= state.words.length) {
          setPlayState(false);
          state.currentIndex = 0;
          updateProgressUI();
          renderActiveWord();
          return;
        }

        renderActiveWord();
        updateProgressUI();

        const currentWord = state.words[state.currentIndex];
        const nextWordDelay = calculateDelayForWord(currentWord);

        state.timerId = setTimeout(() => {
          state.currentIndex++;
          playRSVPEngine();
        }, nextWordDelay);
      }

      // Pause loop manager
      function pauseRSVPEngine() {
        if (state.timerId) {
          clearTimeout(state.timerId);
          state.timerId = null;
        }
      }

      // Word navigation adjustments
      function navigateWords(offset) {
        pauseRSVPEngine();

        state.currentIndex += offset;
        if (state.currentIndex < 0) {
          state.currentIndex = 0;
        } else if (state.currentIndex >= state.words.length) {
          state.currentIndex = state.words.length - 1;
        }

        renderActiveWord();
        updateProgressUI();

        if (state.isPlaying) {
          playRSVPEngine();
        }
      }

      // Set default visual states
      updateAppAppearance();
      updateFontSize(state.fontSize);

    });
  </script>
